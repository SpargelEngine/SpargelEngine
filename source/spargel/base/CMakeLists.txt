spargel_add_library(
    NAME base
    PRIVATE
        allocator.cpp
        panic.cpp
        platform.cpp
        string.cpp
        logging.cpp
        test.cpp
    PRIVATE_POSIX
        platform_posix.cpp
    PRIVATE_ANDROID
        platform_linux.cpp
    PRIVATE_EMSCRIPTEN
        platform_emscripten.cpp
    PRIVATE_LINUX
        platform_linux.cpp
    PRIVATE_MACOS
        platform_mac.cpp
    PRIVATE_WINDOWS
        platform_win.cpp
    DEPS spargel_config
)

spargel_add_library(
    NAME test_main
    PRIVATE
        test_main.cpp
    DEPS base
)

if (SPARGEL_IS_ANDROID)
    target_link_libraries(base PUBLIC m)
endif ()

if (SPARGEL_IS_LINUX)
    target_link_libraries(base PUBLIC
        m
        unwind
        unwind-${CMAKE_SYSTEM_PROCESSOR})
endif ()
